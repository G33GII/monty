#include "monty.h"

/**
 * _free - This Function frees any kind of malloc (integer or string).
 * @_mem: The memory location to be freed.
 * @isdp: An indication for whether it is a array of pointers or not.
 * Return: Void.
 */
void _free(char **dp)
{
    if (*dp == NULL)
    {
        free(dp);
        return;
    }

    free(*dp);
    *dp = NULL;

    _free(dp + 1);
}
