#include "monty.h"

/**
* _lineCount -
* @_b:
* Return: Line count
*/
unsigned int _lineCount(FILE *_b)
{
    char ch;
    unsigned int lines = 0;

    while ((ch = fgetc(_b)) != EOF)
    {
        if (ch == '\n')
            lines++;
    }
    fseek(_b, 0, SEEK_SET);   /* Reset file position to the beginning */
    return (lines);

}
